186LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
187{
188 switch (message)
189 {
190 case WM_COMMAND:
191 {
192 int wmId = LOWORD(wParam);
193 // 메뉴 선택을 구문 분석합니다:
194 switch (wmId)
195 {
196 case IDM_ABOUT:
197 DialogBox(g_hInst, MAKEINTRESOURCE(IDD_ABOUTBOX), hWnd, About);
198 break;
199 case IDM_EXIT:
200 DestroyWindow(hWnd);
201 break;
202 default:
203 return DefWindowProc(hWnd, message, wParam, lParam);
204 }
205 }
206 break;
207
208 case WM_KEYDOWN:
209
210 switch (wParam)
211 {
212 case VK_ESCAPE:
213 DestroyWindow(g_hWnd);
214 break;
215 }
216
217 break;
218
219 case WM_PAINT:
220 {
221 PAINTSTRUCT ps;
222 HDC hdc = BeginPaint(hWnd, &ps);
223 // TODO: 여기에 hdc를 사용하는 그리기 코드를 추가합니다...
224 EndPaint(hWnd, &ps);
225 }
226 break;
227 case WM_DESTROY:
228 PostQuitMessage(0);
229 break;
230 default:
231 return DefWindowProc(hWnd, message, wParam, lParam);
232 }
233 return 0;
234}
